152  // The received frame for euler data : tabrecep (reduced XBus message)):
153  // 1byte : MID = 0x36 => MTData2
154  // 1byte : LEN = 0x0F => 15 data bytes
155  // Data :
156  // 2 bytes : DID = 0x2030 => identifier of the data (euler angles)
157  // 1 byte : DLEN = 0x0C => 12 data (4 bytes for each angle)
158  // 3x4 bytes : 4 Roll, 4 Pitch, 4 Yaw
159  // 1 byte : Checksum
160 
161  // Note that the value of Roll Pitch and Yaw are IEEE float32,
162  // and the bytes are given from the lower weight to the upper weight
163 
164  // We record those data into float values
165  for (uint8_t i=0; i<4; i++) { // memcpy in reverse order!
166  memcpy( ((uint8_t*)(&_eulerAngles.roll)) + 3 - i, &tabrecep[i+5], 1);

168  for (uint8_t i=0; i<4; i++) { // memcpy in reverse order!
169  memcpy( ((uint8_t*)(&_eulerAngles.pitch)) + 3 - i, &tabrecep[i+9], 1);
